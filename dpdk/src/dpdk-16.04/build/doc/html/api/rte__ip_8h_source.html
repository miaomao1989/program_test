167 static inline uint32_t
168 __rte_raw_cksum(const void *buf, size_t len, uint32_t sum)
169 {
170  /* workaround gcc strict-aliasing warning */
171  uintptr_t ptr = (uintptr_t)buf;
172  typedef uint16_t __attribute__((__may_alias__)) u16_p;
173  const u16_p *u16 = (const u16_p *)ptr;
174 
175  while (len >= (sizeof(*u16) * 4)) {
176  sum += u16[0];
177  sum += u16[1];
178  sum += u16[2];
179  sum += u16[3];
180  len -= sizeof(*u16) * 4;
181  u16 += 4;
182  }
183  while (len >= sizeof(*u16)) {
184  sum += *u16;
185  len -= sizeof(*u16);
186  u16 += 1;
187  }
188 
189  /* if length is in odd bytes */
190  if (len == 1)
191  sum += *((const uint8_t *)u16);
192 
193  return sum;
194 }

268 static inline uint16_t
269 rte_ipv4_phdr_cksum(const struct ipv4_hdr *ipv4_hdr, uint64_t ol_flags)
270 {
271  struct ipv4_psd_header {
272  uint32_t src_addr; /* IP address of source host. */
273  uint32_t dst_addr; /* IP address of destination host. */
274  uint8_t zero; /* zero. */
275  uint8_t proto; /* L4 protocol type. */
276  uint16_t len; /* L4 length. */
277  } psd_hdr;
278 
279  psd_hdr.src_addr = ipv4_hdr->src_addr;
280  psd_hdr.dst_addr = ipv4_hdr->dst_addr;
281  psd_hdr.zero = 0;
282  psd_hdr.proto = ipv4_hdr->next_proto_id;
283  if (ol_flags & PKT_TX_TCP_SEG) {
284  psd_hdr.len = 0;
285  } else {
286  psd_hdr.len = rte_cpu_to_be_16(
287  (uint16_t)(rte_be_to_cpu_16(ipv4_hdr->total_length)
288  - sizeof(struct ipv4_hdr)));
289  }
290  return rte_raw_cksum(&psd_hdr, sizeof(psd_hdr));
291 }

354 static inline uint16_t
355 rte_ipv6_phdr_cksum(const struct ipv6_hdr *ipv6_hdr, uint64_t ol_flags)
356 {
357  uint32_t sum;
358  struct {
359  uint32_t len; /* L4 length. */
360  uint32_t proto; /* L4 protocol - top 3 bytes must be zero */
361  } psd_hdr;
362 
363  psd_hdr.proto = (ipv6_hdr->proto << 24);
364  if (ol_flags & PKT_TX_TCP_SEG) {
365  psd_hdr.len = 0;
366  } else {
367  psd_hdr.len = ipv6_hdr->payload_len;
368  }
369 
370  sum = __rte_raw_cksum(ipv6_hdr->src_addr,
371  sizeof(ipv6_hdr->src_addr) + sizeof(ipv6_hdr->dst_addr),
372  0);
373  sum = __rte_raw_cksum(&psd_hdr, sizeof(psd_hdr), sum);
374  return __rte_raw_cksum_reduce(sum);
375 }
