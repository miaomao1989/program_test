179 /* Search given slave array to find possition of given id.
180  * Return slave pos or slaves_count if not found. */
181 static inline uint8_t
182 find_slave_by_id(uint8_t *slaves, uint8_t slaves_count, uint8_t slave_id) {
183 
184  uint8_t pos;
185  for (pos = 0; pos < slaves_count; pos++) {
186  if (slave_id == slaves[pos])
187  break;
188  }
189 
190  return pos;
191 }
