102 rte_version(void)
103 {
104  static char version[32];
105  if (version[0] != 0)
106  return version;
107  if (strlen(RTE_VER_SUFFIX) == 0)
108  snprintf(version, sizeof(version), "%s %d.%02d.%d",
109  RTE_VER_PREFIX,
110  RTE_VER_YEAR,
111  RTE_VER_MONTH,
112  RTE_VER_MINOR);
113  else

115  RTE_VER_PREFIX,

117  RTE_VER_MONTH,
118  RTE_VER_MINOR,
119  RTE_VER_SUFFIX,
120  RTE_VER_RELEASE < 16 ?
121  RTE_VER_RELEASE :
122  RTE_VER_RELEASE - 16);
123  return version;
